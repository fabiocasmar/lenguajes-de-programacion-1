def initialize m
        self.k = m.keys
        self.v = m.values.inject(0, :+)
        self.gen = Random.new(SEED)
        self.x = []
        self.sum = 0
        self.l = m.each do |k,v| 
                v.times {self.x.push(k)}
                self.sum+=v
        end
end
