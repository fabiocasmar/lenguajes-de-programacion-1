La función bfs realiza la ejecución del bfs de manera genérica, haciendo uso de la función each de la clase, del objeto.

# File bfs.rb, line 20
def bfs(start)
    q          = []
    q.push(start)
    visit      = []
    pushear_nodo = lambda { |x| q.push(x) } 
    while(q.size != 0)
        n = q.shift
        if !visit.include? n
            n.each pushear_nodo 
            yield n 
            visit.push(n)
        end
    end
end

def find(start, predicate)
    raise "start debe tener modulo bfs." unless start.respond_to? :bfs
    start.bfs(start) { |n| return n if predicate.call(n) } 
end

La función path recibe un nodo desde el cual se ejecutará bfs y un predicado. Se devolverá el camino desde el nodo start, hasta el primer nodo que cumpla el predicado.

# File bfs.rb, line 48
def path(start, predicate)
    raise "start debe tener modulo bfs." unless start.respond_to? :bfs
    f = { start => nil }
    start.bfs(start) do |n|
        f.keys.each do |x|
            if x == n
                n = x
            end
        end
  
        if predicate.call(n)
            pa = []
            while !(n.nil?) do
                pa.unshift(n)
                n = f[n]
            end
            return pa
        end
        n.each lambda { |h| f[h] = n }   
      end   
end

La función walk recibe un nodo desde el cual se ejecutará bfs y un action que se ejecutará sobre cada nodo. Se devolverá una lista con todos los nodos visitado.

# File bfs.rb, line 75
def walk (start, action)
    raise "start debe tener modulo bfs." unless start.respond_to? :bfs
    v = []
    start.bfs(start) do |n| 
        action.call(n)
        v.push(n)
    end 
    return v
end
